REM MISSclr goes through series of MISS w/color indicating total
CLS
SCREEN 0, 1
x = 0
DO
d = 0: c = 2: x = x + 1
DO UNTIL c * 2 > x
a = 0
two: a = a + 1
IF x MOD c ^ a = 0 THEN d = d + 1: GOTO two
c = c + 1
LOOP
g = d MOD 3
COLOR g, 0, 8
IF d MOD 2 = 0 THEN PRINT "Û";  ELSE PRINT "Ü";
LOOP
